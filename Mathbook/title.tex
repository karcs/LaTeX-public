% This is file `title.sty',

% 0 REQUIRED PACKAGES

% 1 GET AND SET TITLE

% to get and set the (current) title of a layer or environment
% it is stored in \csname #1title \endcsname

% set title
\def\settitle#1#2{%
    \expandafter\def\csname #1title\endcsname{#2}%
}%

% get title
\def\gettitle#1{%
    \csname #1title\endcsname%
}%

% get title of current layer
\def\title{%
    \expandafter\csname\layer title\endcsname%
}%
