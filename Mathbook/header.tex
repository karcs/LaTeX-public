% HEADERS OF LAYERS AND ENVIRONMENTS

% set header of layer or environment #1
\def\setheader#1#2{%
    \expandafter\def\csname #1header\endcsname{#2}%
}%

% get header of layer or environment #1
\def\getheader#1{%
    \begingroup%
    \expandafter\csname #1headerpreskip\endcsname%
    \expandafter\csname #1headerfont\endcsname%
    \expandafter\csname #1header\endcsname%
    \expandafter\csname #1headerpostskip\endcsname%
    \endgroup%
}%

% get the current header
\def\header{%
    \begingroup%
    \expandafter\csname\layer 1headerpreskip\endcsname%
    \expandafter\csname\layer headersize\endcsname%
    \expandafter\csname\layer headerfont\endcsname%
    \expandafter\csname\layer header\endcsname%
    \expandafter\csname\layer headerpostskip\endcsname%
    \endgroup%
}%

% HEADER FONT

% set header font of layer or environment #1

% get the font of current header
\def\headerfont{%
    \expandafter\csname\layer header\endcsname%
}%
